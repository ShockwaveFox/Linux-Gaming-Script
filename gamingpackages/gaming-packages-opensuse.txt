# gaming-packages-opensuse.txt

# ===  Add Packman ===
sudo zypper addrepo -f -p 90 'https://ftp.gwdg.de/pub/linux/misc/packman/suse/openSUSE_Tumbleweed/' packman

# ===  Full upgrade + switch packages to Packman ===  (very important for Mesa)
sudo zypper dup --allow-vendor-change

# ===  Install everything === 
sudo zypper install $(cat gaming-packages-opensuse.txt)
