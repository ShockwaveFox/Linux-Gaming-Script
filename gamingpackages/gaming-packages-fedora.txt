# Add the ShadowBlip InputPlumber COPR repository
sudo dnf copr enable shadowblip/InputPlumber

# Install InputPlumber
sudo dnf install inputplumber

# Enable and start the service (important for your GameSir G7 Pro)
systemctl --user enable --now inputplumber

# Optional: Enable suspend service (helps with resume behavior)
sudo systemctl enable inputplumber-suspend.service
